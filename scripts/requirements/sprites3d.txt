#   uv venv .venv-sprites --python 3.10
#   uv pip install --python .venv-sprites -r scripts/requirements/sprites3d.txt
#   PYOPENGL_PLATFORM=egl .venv-sprites/bin/python scripts/generate_planet_sprites.py
#
# Or just:  uv run scripts/generate_planet_sprites.py
